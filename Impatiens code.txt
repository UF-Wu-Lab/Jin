#!/bin/bash
#SBATCH --job-name=HiFi_FastQC
#SBATCH --qos=xingbo.wu-b
#SBATCH --ntasks=1
#SBATCH --cpus-per-task=10
#SBATCH --time=23:00:00
#SBATCH --mem=80G

# Load modules
module load samtools/1.17
module load fastqc/0.12.1

# Create FastQC output directory in current dir
mkdir -p fastqc_reports

# Output combined FASTQ filename
combined_fastq="Impatiens_HiFi_combined.fastq.gz"

echo "Converting and combining BAM files into $combined_fastq ..."

# Loop through BAM files and append to combined gzipped FASTQ
for bam in /orange/xingbo.wu/Rawdata/Impatiens/RawData/rawreads/*.bam; do
    echo "Processing $bam ..."
    samtools fastq $bam | gzip >> $combined_fastq
done

echo "Running FastQC on $combined_fastq ..."
fastqc -t 10 -o fastqc_reports $combined_fastq

echo "Done. FastQC report is in ./fastqc_reports/"

#### OrthoFinder clean protein fasta header

#!/bin/bash

# List of FASTA files
files=(
  "Impatiens_glandulifera.fasta"
  "Impatiens_noli-tangere.fasta"
  "Impatiens_parviflora.fasta"
  "Impatiens_walleriana.fasta"
)

# Loop through each file
for f in "${files[@]}"; do
    # Extract species name (without .fasta) and ensure it contains "Impatiens"
    species="${f%.fasta}"
    if [[ "$species" != Impatiens* ]]; then
        species="Impatiens_$species"
    fi
    
    # Append species name to each header
    sed 's/^>\(.*\)/>\1|'"$species"'/' "$f" > "${species}_modified.fasta"
    echo "Processed $f → ${species}_modified.fasta"
done

#!/bin/bash
#SBATCH --job-name=cdhit_batch

# Set identity threshold (adjust if necessary)
IDENTITY_THRESHOLD=0.99

# Loop through all .aa files in the current folder
for file in *.fasta; do
    # Extract the filename without the extension
    base_name=$(basename "$file" .fasta)

    # Run CD-HIT to retain the longest isoform for each protein

#### merge reads  (Merges all Gblocks-trimmed files. Removes spaces. Removes line breaks in sequences. Concatenates sequences by species/cultivar and writes one FASTA per species. Combine into one fasta for iqtree)

#!/bin/bash

# --- Step 1: Merge all GBlocks-trimmed files ---
cat ./*-gb > combined.fasta

# --- Step 2: Remove all spaces in sequences ---
sed 's/ //g' combined.fasta > combined_no_spaces.fasta

# --- Step 3: Merge multi-line sequences into a single line per sequence ---
awk '!/^>/ {printf "%s", $0; n=1; next} n {print ""; n=0} 1' combined_no_spaces.fasta > combined_no_spaces_linebreaks.fasta

# --- Step 4: Separate sequences by species/cultivar and concatenate ---
input="combined_no_spaces_linebreaks.fasta"

awk '
BEGIN { OFS="" }
# Header line
/^>/ {
    if (seq != "" && sp != "") {
        seqs[sp] = seqs[sp] seq
    }
    seq = ""

    # Extract species/cultivar name
    if (match($0, /\|([A-Za-z_]+)/, a)) {
        sp = a[1]
    } else if (match($0, /^>([^_]+)_g[0-9]+/, b)) {
        sp = b[1]
    } else {
        sp = substr($0, 2)  # fallback: raw header without ">"
    }
    next
}
# Sequence lines
{
    seq = seq $0
}
END {
    if (seq != "" && sp != "") {
        seqs[sp] = seqs[sp] seq
    }

    # Write each species to its own file
    for (s in seqs) {
        out = s "_concatenated.fasta"
        print ">" s "\n" seqs[s] > out
    }
}' "$input"

echo "✅ GBlocks merge and concatenation complete. Each species has its own FASTA."

# --- Step 5: Merge all concatenated species FASTA files into one file for IQ-TREE
 
# Create folder if not exists
mkdir -p iqtree

cat *_concatenated.fasta > ./iqtree/all_species_concatenated.fasta

#### CAFE 

### format gene count file and filter large families (>100 genes) to avoid unstable likelihoods


#!/usr/bin/env bash
set -euo pipefail

input="Orthogroups.GeneCount.tsv"
tabbed="Orthogroups.GeneCount.tab.tsv"
no_vitis="Orthogroups.GeneCount.noVitis.tsv"
modified="Orthogroups.GeneCount.modified.tsv"
filtered="Orthogroups.GeneCount.filtered.tsv"

# ===================================================
# Step 0: Normalize whitespace to single TAB
# ===================================================
awk '{$1=$1}1' OFS='\t' "$input" > "$tabbed"

# ===================================================
# Step 1: Remove Vitis_vinifera.clean column
# ===================================================
awk -F'\t' -v OFS='\t' '
NR==1 {
    skip_col=0
    for (i=1; i<=NF; i++) {
        if ($i == "Vitis_vinifera.clean") skip_col=i
    }
}
{
    first=1
    for (i=1; i<=NF; i++) {
        if (i != skip_col) {
            if (!first) printf OFS
            printf "%s", $i
            first=0
        }
    }
    print ""
}' "$tabbed" > "$no_vitis"

echo "✅ Removed Vitis_vinifera.clean column."

# ===================================================
# Step 2: Add Desc, remove Total, trim headers
# ===================================================
awk -F'\t' -v OFS='\t' '
NR==1 {
    printf "Desc"
    printf OFS "%s", $1
    for (i=2; i<NF; i++) {
        split($i, a, "_")
        if (length(a) >= 2)
            printf OFS "%s_%s", a[1], a[2]
        else
            printf OFS "%s", $i
    }
    print ""
    next
}
{
    printf "(null)"
    for (i=1; i<NF; i++)
        printf OFS "%s", $i
    print ""
}' "$no_vitis" > "$modified"

echo "✅ Created cleaned file with Desc column -> $modified"

# ===================================================
# Step 3: Run clade_and_size_filter.py
# ===================================================
python clade_and_size_filter.py \
    -i "$modified" \
    -o "$filtered" \
    -s 

echo "✅ Filtering done."
echo "Filtered file: $filtered"

library(tidyverse)
library(VennDiagram)
library(RColorBrewer)
library(grid)

# ----------------------
# Step 1: Read and process data
# ----------------------
gene_counts <- read.delim("Orthogroups.GeneCount.tsv", stringsAsFactors = FALSE)
impatiens_cols <- grep("Impatiens", colnames(gene_counts))
imp_counts <- gene_counts[, impatiens_cols]

# Convert to presence/absence
imp_binary <- ifelse(imp_counts > 0, 1, 0)
rownames(imp_binary) <- gene_counts$Orthogroup

# Clean species names
clean_names <- colnames(imp_binary) %>%
  str_replace("_modified_longest_isoform", "") %>%
  str_replace("_", " ") %>%
  str_replace("\\.", "-")

clean_names <- sapply(clean_names, function(x) {
  parts <- unlist(strsplit(x, " "))
  paste0(toupper(substr(parts[1],1,1)), substr(parts[1],2,nchar(parts[1])), " ",
         tolower(parts[2]))
})
colnames(imp_binary) <- clean_names

# ----------------------
# Step 2: Prepare list for Venn
# ----------------------
venn_list <- lapply(colnames(imp_binary), function(sp) {
  rownames(imp_binary)[imp_binary[, sp] == 1]
})
names(venn_list) <- colnames(imp_binary)

colors <- brewer.pal(length(venn_list), "Set2")

# ----------------------
# Step 3: Draw Venn diagram without default labels
# ----------------------

  x = venn_list,
  filename = NULL,
  fill = colors,
  alpha = 0.6,
  cex = 2.7,             # counts inside circles
  fontface = "plain",
  fontfamily = "Arial",
  cat.cex = 0,           # remove default labels

  main = NULL
)

from Bio import SeqIO

ref_file = "$REF"
out_file = "$OUTDIR/Impatiens_GC_${WINDOW}.bed"
window = int("$WINDOW")

with open(out_file, 'w') as out:
    for record in SeqIO.parse(ref_file, "fasta"):
        chrom = record.id
        seq = str(record.seq).upper()
        for i in range(0, len(seq), window):

            gc = window_seq.count('G') + window_seq.count('C')
            if len(window_seq) > 0:
                gc_percent = gc / len(window_seq) * 100
                out.write(f"{chrom}\t{i}\t{i+len(window_seq)}\t{gc_percent:.2f}\n")
EOF

SV_VCF="Impatiens_HiFi.sniffles.filtered.vcf.gz"

# Output files for Circos
SNP_OUT="circos_snps.txt"

INV_OUT="inversions.circos.txt"

# Clear output files
> $SNP_OUT
> $INDEL_OUT
> $INV_OUT

### ------------------------------
### 1. Convert SNPs and INDELs to Circos format
### ------------------------------
# SNPs: single-base variants
# INDELs: insertion/deletion variants
# Circos format: chromosome, start (0-based), end, weight

zcat $VCF | awk 'BEGIN{OFS="\t"}
    !/^#/ {
        chrom=$1
        pos=$2
        ref=$4
        alt=$5
        # SNP: single-base change
        if(length(ref)==1 && length(alt)==1){
            print chrom, pos-1, pos, 1 >> "'$SNP_OUT'"
        }
        # INDEL: length difference between ref and alt
        else if(length(ref)!=length(alt)){
            print chrom, pos-1, pos+length(ref)-1, 1 >> "'$INDEL_OUT'"
        }
    }'

### ------------------------------
### 2. Convert inversions (SVs) to Circos format
### ------------------------------
# Only inversions are kept
# Circos format: chromosome, start, end

zcat $SV_VCF | \
awk '$0 !~ /^#/ && $8 ~ /SVTYPE=INV/ {
    split($8, a, ";");
    for(i in a) if(a[i] ~ /^END=/) end_pos=substr(a[i],5);
    print $1, $2, end_pos
}' > $INV_OUT

awk '{print $1, $2, $3, 1}' OFS="\t" inversions.circos.txt > inversions.circos.fixed.txt

mv inversions.circos.fixed.txt inversions.circos.txt

#!/bin/bash
#SBATCH --job-name=Imp_normbed
#SBATCH --qos=xingbo.wu-b
#SBATCH --cpus-per-task=4
#SBATCH --mem=16G
#SBATCH --time=04:00:00

# -----------------------------
# Load modules
# -----------------------------
ml bedtools/2.30.0

# -----------------------------
# Parameters
# -----------------------------
WINDOW=1000000
GENOME_FILE="./genome.sizes"   # chromosome-only genome file
OUTDIR="."
VARIANT_DIR="./Variant"

# -----------------------------
# Make windows
# -----------------------------
bedtools makewindows -g $GENOME_FILE -w $WINDOW > $OUTDIR/windows.bed

# -----------------------------
# Calculate gene density
# -----------------------------
bedtools intersect -a $OUTDIR/windows.bed -b ./Gene/Impatiens_genes.bed -c > ./gene_density_${WINDOW}.bed

# -----------------------------
# Calculate TE density
# -----------------------------
bedtools intersect -a $OUTDIR/windows.bed -b ./TE/Impatiens_TE.bed -c > ./TE_density_${WINDOW}.bed

# -----------------------------
# Calculate SNP density
# -----------------------------
bedtools intersect -a $OUTDIR/windows.bed -b $VARIANT_DIR/circos_snps.txt -c > ./snp_density_${WINDOW}.bed

# -----------------------------
# Calculate INDEL density
# -----------------------------
bedtools intersect -a $OUTDIR/windows.bed -b $VARIANT_DIR/circos_indels.txt -c > ./indel_density_${WINDOW}.bed

# -----------------------------
# Normalize densities (0-1)
# -----------------------------
for f in $OUTDIR/*_density_${WINDOW}.bed; do
    max=$(awk '{if($4>m)m=$4}END{print m}' $f)
    awk -v max=$max '{print $1, $2, $3, $4, $4/max}' OFS="\t" $f > ${f%.bed}.norm.bed
done

# -----------------------------
# Prepare Circos-ready bed files
# -----------------------------
for f in $OUTDIR/*norm.bed; do
    awk '{print $1, $2, $3, $5}' OFS="\t" "$f" > "${f%.norm.bed}.circos.bed"
done

import pandas as pd

# Read BED file
bed = pd.read_csv("Impatiens_walleriana.bed", sep="\t", header=None, names=["chr","start","end","gene","score","strand"])

# Merge duplicate genes: keep min start and max end
bed_merged = bed.groupby("gene").agg({"chr":"first", "start":"min", "end":"max"}).reset_index()

# Create dictionary
gene_dict = bed_merged.set_index("gene")[["chr","start","end"]].to_dict("index")

# Read anchors
anchors = pd.read_csv("Impatiens_walleriana.Impatiens_walleriana.anchors.simple", sep="\t", header=None)

links = []
for idx, row in anchors.iterrows():
    geneA = row[0]
    geneB = row[2]
    if geneA in gene_dict and geneB in gene_dict:
        chrA = gene_dict[geneA]["chr"]
        chrB = gene_dict[geneB]["chr"]
        startA = gene_dict[geneA]["start"]
        endA = gene_dict[geneA]["end"]
        startB = gene_dict[geneB]["start"]
        endB = gene_dict[geneB]["end"]
        links.append([chrA, startA, endA, chrB, startB, endB])

# Save links for circos plotting
links_df = pd.DataFrame(links)
links_df.to_csv("synteny_links.txt", sep="\t", header=False, index=False)

### fix transcript ID if renamed by JCVI with fix_bed.sh

#!/bin/bash

# Directory where your BED files are
BED_DIR="./gtf"
# Directory where your GFF3 files are
GFF_DIR="./gtf"

# List of species (BED and GFF3 files)
species=("Impatiens_glandulifera" "Impatiens_noli-tangere" "Impatiens_parviflora" "Impatiens_walleriana")

for sp in "${species[@]}"; do
    BED_FILE="${GFF_DIR}/${sp}.bed"
    GFF_FILE="${GFF_DIR}/${sp}.gff3"
    OUT_FILE="./${sp}.bed"

    echo "Processing $sp ..."

    awk -F'\t' 'BEGIN{
        OFS="\t"
    }
    FNR==NR && $3=="transcript" {
        # Store mRNA coords -> transcript_id
        match($9,/ID=([^;]+)/,a)
        key=$1":"$4":"$5
        tid[key]=a[1]
        next
    }
    FNR!=NR {
        # BED: convert start to 1-based for lookup
        key=$1":"$2+1":"$3
        if(key in tid) {
            $4=tid[key]
        }
        print
    }' "$GFF_FILE" "$BED_FILE" > "$OUT_FILE"

    echo "Saved corrected BED to $OUT_FILE"
done

        gunzip $DBDIR/uniprot_sprot.fasta.gz

#### KEGG gene finder (A Bash script to extract genes from Orthogroups.tsv for specified species and orthogroup lists)

#!/bin/bash

# Usage:
# bash gene_finder.sh Orthogroups.tsv "Impatiens_walleriana_modified_longest_isoform" expanded_family.txt output.txt

set -euo pipefail

orthogroup_file=$1
species_name=$2
family_list=$3
output_file=$4

# Check input files exist
if [[ ! -f "$orthogroup_file" ]]; then
    echo "Error: Orthogroup file '$orthogroup_file' not found."
    exit 1
fi

if [[ ! -f "$family_list" ]]; then
    echo "Error: Family list file '$family_list' not found."
    exit 1
fi

# Convert DOS line endings to Unix
dos2unix "$orthogroup_file" "$family_list"

# Find column number for the species (1-based)
col=$(head -1 "$orthogroup_file" | tr -s ' ' '\t' | tr '\t' '\n' | grep -n "^$species_name$" | cut -d':' -f1)

if [[ -z "$col" ]]; then
    echo "Error: Species column '$species_name' not found in header."
    exit 1
fi

echo "Species column '$species_name' found at column $col."

# Make output file empty
> "$output_file"

# Read each orthogroup from the family list (handle Windows line endings)
while IFS=$'\r\n' read -r og; do
    [[ -z "$og" ]] && continue  # skip empty lines

    awk -F'\t' -v col="$col" -v og="$og" '
    $1 == og {
        # replace commas with space, then split on spaces
        gsub(/,/, " ", $col)
        n = split($col, genes, /[ ]+/)
        for (i=1; i<=n; i++) {
            gsub(/^ +| +$/,"",genes[i])   # trim spaces
            if (genes[i] != "") print genes[i]
        }
    }' "$orthogroup_file" >> "$output_file"

done < "$family_list"

echo "Done! Genes written to: $output_file"

#### extract_genes_OG_KO.py (put ko of interest in ko.txt, extract genes from eggNOG annotation and retrieve OG number from Orthogroups.tsv)

#!/usr/bin/env python3
import pandas as pd
import glob

# -------- Parameters --------
GENE_PATTERNS = "ko.txt"          # file with patterns to search, one per line
ANNOT_PATTERN = "*_eggnog.emapper.annotations"
ORTHOFILE = "Orthogroups.tsv"
OUTPUT = "genes_OG_table.tsv"

# -------- Step 1: Load gene patterns --------
with open(GENE_PATTERNS) as f:
    patterns = [line.strip() for line in f if line.strip()]
print(f"Loaded {len(patterns)} patterns to search.")

# -------- Step 2: Build gene -> OG mapping --------
orthos = pd.read_csv(ORTHOFILE, sep="\t")
gene2OG = {}
for idx, row in orthos.iterrows():
    OG = row.iloc[0]
    for col in row[1:]:
        if pd.notna(col):
            for gene in str(col).split(","):  # handle multiple genes per cell
                gene2OG[gene.strip()] = OG  # keep full gene ID with species
print(f"Mapped {len(gene2OG)} genes to OGs.")

# -------- Step 3: Search annotation files --------
results = []
for annot_file in glob.glob(ANNOT_PATTERN):
    species = annot_file.split("_eggnog")[0]
    print(f"Processing {species}...")

    with open(annot_file) as f:
        for line in f:
            if line.startswith("#"):
                continue
            for pattern in patterns:
                if pattern in line:
                    gene_name = line.strip().split("\t")[0]  # keep full gene ID
                    OG = gene2OG.get(gene_name, "NA")        # lookup using full gene ID
                    results.append([gene_name, OG, pattern])
                    break  # stop after first pattern match per line

# -------- Step 4: Save results --------
if results:
    df = pd.DataFrame(results, columns=["Gene_name", "OG_number", "Pattern"])
    df.to_csv(OUTPUT, sep="\t", index=False)
    print(f"Done! {len(results)} rows saved to {OUTPUT}")
else:
    print("No matches found.")

#!/usr/bin/env python3
import pandas as pd

# -------- Parameters --------
INPUT = "genes_OG_table.tsv"   # your gene-level table
OUTPUT = "OG_count_table.tsv"  # output OG x species count table

# -------- Step 1: Load gene table --------
df = pd.read_csv(INPUT, sep="\t")
print(f"Loaded {len(df)} rows from {INPUT}")

# -------- Step 2: Extract species from Gene_name --------
# Assuming format: gene_id|species
df['Species'] = df['Gene_name'].apply(lambda x: x.split("|")[1] if "|" in x else x)

# -------- Step 3: Group by OG and Species to count genes --------
count_df = df.groupby(['OG_number', 'Species']).size().reset_index(name='Count')

# -------- Step 4: Pivot table to have species as columns --------
og_table = count_df.pivot(index='OG_number', columns='Species', values='Count').fillna(0).astype(int)

# Optional: sort rows and columns
og_table = og_table.sort_index()
og_table = og_table.reindex(sorted(og_table.columns), axis=1)

# -------- Step 5: Save to file --------
og_table.to_csv(OUTPUT, sep="\t")
print(f"OG count table saved to {OUTPUT}")

/blue/xingbo.wu/jinli/impatiens/genus/EDTA/genome/Impatiens_parviflora.fasta

echo "    Using $SLURM_CPUS_PER_TASK threads"
echo "    Output directory: $OUTDIR"

# === Step 1: Run EDTA (sensitive mode) ===
EDTA.pl \
  --genome "$GENOME" \
  --species others \
  --step all \
  --threads $SLURM_CPUS_PER_TASK \
  --anno 1 \
  --evaluate 1 \
  --force 1

# === Step 2: Define file paths ===
GFF3="${OUTDIR}/${BASENAME}.fasta.mod.EDTA.TEanno.gff3"
BED="${OUTDIR}/${BASENAME}.fasta.mod.EDTA.TEanno.struc.bed"
PASSLIST="${OUTDIR}/${BASENAME}.fasta.mod.EDTA.TEanno.LTR.pass.list"
RMOUT="${OUTDIR}/${BASENAME}.fasta.mod.EDTA.TEanno.out"

# === Step 3: Generate BED from GFF3 ===
if [[ ! -f "$BED" ]]; then
    echo "Generating BED from GFF3..."
    perl /apps/edta/2.2.2/share/EDTA/bin/gff2bed.pl "$GFF3" structural > "$BED"
fi

# === Step 4: Generate LTR pass list for LAI ===
echo "Generating LTR pass list..."
grep LTR "$BED" | grep struc | awk '{print $1":"$2".."$3"\t"$4}' > "$PASSLIST"

if [[ ! -s "$PASSLIST" ]]; then
    echo "Error: pass list is empty ($PASSLIST)"
    exit 1
fi

# === Step 5: Ensure .out file exists ===
if [[ ! -f "$RMOUT" ]]; then
    echo "EDTA .out file not found, generating pseudo .out from BED..."
    perl -nle 'my ($chr,$s,$e,$anno,$dir,$supfam) = (split)[0,1,2,3,8,12];
               print "10000 0.001 0.001 0.001 $chr $s $e NA $dir $anno $supfam"' \
        "$BED" > "$RMOUT"
fi


echo ">>> Finished processing $GENOME"
